CREATE TABLE authors (
    author_id INT AUTO_INCREMENT PRIMARY KEY,
    author_name VARCHAR(255) NOT NULL
);

CREATE TABLE genres (
    genre_id INT AUTO_INCREMENT PRIMARY KEY,
    genre_name VARCHAR(255) NOT NULL
);

CREATE TABLE books (
    book_id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    publication_year YEAR,
    author_id INT NOT NULL,
    genre_id INT NOT NULL,
    FOREIGN KEY (author_id) REFERENCES authors(author_id),
    FOREIGN KEY (genre_id) REFERENCES genres(genre_id)
);

CREATE TABLE users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL
);


CREATE TABLE borrowed_books (
    borrow_id INT AUTO_INCREMENT PRIMARY KEY,
    book_id INT NOT NULL,
    user_id INT NOT NULL,
    borrow_date DATE NOT NULL,
    return_date DATE,
    FOREIGN KEY (book_id) REFERENCES books(book_id),
    FOREIGN KEY (user_id) REFERENCES users(user_id)
);

Під час виконання завдання я змінила декілька операторів JOIN у запиті до таблиць бази даних. Початковий запит використовував лише INNER JOIN, і результат містив 518 рядків. Це повністю відповідало кількості рядків у таблиці order_details, яка є базовою таблицею для з’єднання.

Далі я змінила деякі INNER JOIN на LEFT JOIN та RIGHT JOIN. Незважаючи на це, результат знову показав 518 рядків. Це сталося тому, що в наявних даних усі зовнішні ключі коректні та кожен рядок у order_details має відповідні записи у пов’язаних таблицях (orders, products, customers, suppliers, categories тощо).

LEFT JOIN не відкидає рядки з лівої таблиці, а RIGHT JOIN – з правої. Але оскільки всі дані узгоджені (немає «висячих» записів і немає зовнішніх ключів, які не мають відповідної пари), такі JOIN-и не збільшують і не зменшують кількість рядків. Вони просто додають NULL-значення для тих випадків, де відповідних рядків могло б не бути. У цій базі таких ситуацій не виникає.

Таким чином, кількість рядків не змінилася після заміни INNER на LEFT/RIGHT JOIN через те, що всі таблиці ідеально пов'язані між собою і кожен запис має відповідність у зв’язаних таблицях. Якби в базі існували записи без відповідних зовнішніх ключів, LEFT JOIN збільшив би кількість рядків, а INNER JOIN – зменшив би. У нашому випадку цього не відбулося, тому результат залишився незмінним (518 рядків).
